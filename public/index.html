-- Courses table CREATE TABLE courses ( id BIGSERIAL PRIMARY KEY, title TEXT, description TEXT, price INTEGER, duration TEXT, units INTEGER, icon TEXT, is_published BOOLEAN DEFAULT true, "order" INTEGER, modules JSONB ); -- Quiz questions table CREATE TABLE quiz_questions ( id BIGSERIAL PRIMARY KEY, question_text TEXT, option_a TEXT, option_b TEXT, option_c TEXT, option_d TEXT, correct_option CHAR(1), explanation TEXT ); -- Enrollments table CREATE TABLE enrollments ( id BIGSERIAL PRIMARY KEY, user_id UUID, course_id BIGINT, course_name TEXT, price INTEGER, status TEXT DEFAULT 'active', payment_method TEXT, transaction_id TEXT, enrolled_at TIMESTAMPTZ ); -- Profiles table CREATE TABLE profiles ( id UUID PRIMARY KEY, email TEXT, full_name TEXT, role TEXT DEFAULT 'learner' );
